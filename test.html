Вот переписанный docker-compose.yml файл, включающий GraphQL Playground и RedisInsight, а также настройку для сохранения данных баз данных при перезагрузке контейнеров.


 

Файл docker-compose.yml


 

version: '3.8'


 

services:

  # Drupal

  drupal:

    image: drupal:10

    container_name: drupal

    ports:

      - "8080:80"

    volumes:

      - ./drupal:/var/www/html

    environment:

      DRUPAL_DB_HOST: drupal_db

      DRUPAL_DB_NAME: drupal

      DRUPAL_DB_USER: drupal

      DRUPAL_DB_PASSWORD: drupal

    depends_on:

      - drupal_db

      - redis


 

  drupal_db:

    image: mysql:8.0

    container_name: drupal_db

    environment:

      MYSQL_ROOT_PASSWORD: root

      MYSQL_DATABASE: drupal

      MYSQL_USER: drupal

      MYSQL_PASSWORD: drupal

    ports:

      - "3306:3306"

    volumes:

      - drupal_db_data:/var/lib/mysql


 

  phpmyadmin:

    image: phpmyadmin/phpmyadmin

    container_name: phpmyadmin

    ports:

      - "8081:80"

    environment:

      PMA_HOST: drupal_db

      MYSQL_ROOT_PASSWORD: root


 

  # Redis

  redis:

    image: redis:alpine

    container_name: redis

    ports:

      - "6379:6379"

    volumes:

      - redis_data:/data


 

  redisinsight:

    image: redisinsight:latest

    container_name: redisinsight

    ports:

      - "8001:8001"


 

  # GraphQL

  graphql:

    image: node:16-alpine

    container_name: graphql

    working_dir: /app

    volumes:

      - ./graphql:/app

    ports:

      - "4000:4000"

    command: ["npm", "start"]

    depends_on:

      - drupal

      - redis


 

  graphql_playground:

    image: graphql-playground

    container_name: graphql_playground

    ports:

      - "4001:3000"

    environment:

      GRAPHQL_PLAYGROUND_ENDPOINT: http://graphql:4000


 

  # WordPress

  wordpress:

    image: wordpress:latest

    container_name: wordpress

    ports:

      - "8082:80"

    volumes:

      - ./wordpress:/var/www/html

    environment:

      WORDPRESS_DB_HOST: wordpress_db

      WORDPRESS_DB_NAME: wordpress

      WORDPRESS_DB_USER: wordpress

      WORDPRESS_DB_PASSWORD: wordpress

    depends_on:

      - wordpress_db

      - redis


 

  wordpress_db:

    image: postgres:latest

    container_name: wordpress_db

    environment:

      POSTGRES_USER: wordpress

      POSTGRES_PASSWORD: wordpress

      POSTGRES_DB: wordpress

    ports:

      - "5433:5432"

    volumes:

      - wordpress_db_data:/var/lib/postgresql/data


 

  # Symfony

  symfony:

    image: php:8.2-apache

    container_name: symfony

    ports:

      - "8083:80"

    volumes:

      - ./symfony:/var/www/html

    environment:

      SYMFONY_DB_HOST: symfony_db

      SYMFONY_DB_NAME: symfony

      SYMFONY_DB_USER: symfony

      SYMFONY_DB_PASSWORD: symfony

    depends_on:

      - symfony_db

      - redis


 

  symfony_db:

    image: postgres:latest

    container_name: symfony_db

    environment:

      POSTGRES_USER: symfony

      POSTGRES_PASSWORD: symfony

      POSTGRES_DB: symfony

    ports:

      - "5434:5432"

    volumes:

      - symfony_db_data:/var/lib/postgresql/data


 

  # React.js

  react:

    build:

      context: ./react

    container_name: react

    ports:

      - "3000:3000"

    volumes:

      - ./react:/app

    command: ["npm", "start"]


 

  # Next.js

  nextjs:

    build:

      context: ./nextjs

    container_name: nextjs

    ports:

      - "3001:3000"

    volumes:

      - ./nextjs:/app

    command: ["npm", "run", "dev"]


 

  # pgAdmin

  pgadmin:

    image: dpage/pgadmin4

    container_name: pgadmin

    ports:

      - "8084:80"

    environment:

      PGADMIN_DEFAULT_EMAIL: admin@admin.com

      PGADMIN_DEFAULT_PASSWORD: admin

    depends_on:

      - wordpress_db

      - symfony_db


 

volumes:

  drupal_db_data:

  wordpress_db_data:

  symfony_db_data:

  redis_data:


 

Изменения для сохранения данных:


 

1. Tomcat для Redis:

Добавлен volume redis_data:/data для сохранения данных Redis.

2. Базы данных MySQL и PostgreSQL:

Используются volumes drupal_db_data, wordpress_db_data, symfony_db_data для сохранения данных MySQL и PostgreSQL.

3. GraphQL Playground:

Добавлен отдельный сервис graphql_playground для визуального тестирования GraphQL-запросов. Он подключен к GraphQL-серверу через GRAPHQL_PLAYGROUND_ENDPOINT.


 

Как запустить:


 

1. Создайте директории:


 

mkdir drupal wordpress symfony graphql react nextjs


 


 

2. Настройте приложения в этих папках.

3. Запустите Docker Compose:


 

docker-compose up -d


 


 

4. Проверьте доступность сервисов:

• Drupal: http://localhost:8080

• phpMyAdmin: http://localhost:8081

• WordPress: http://localhost:8082

• Symfony: http://localhost:8083

• RedisInsight: http://localhost:8001

• GraphQL Playground: http://localhost:4001


 

Преимущества:


 

1. Сохранение данных:

Все данные сохраняются при перезапуске контейнеров.

2. Визуальный интерфейс:

RedisInsight и GraphQL Playground обеспечивают удобную работу с Redis и GraphQL.

3. Масштабируемость:

Возможность добавить новые приложения или модули без значительных изменений.
